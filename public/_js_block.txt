  async function handleImg(file) {

  }

  function resizeImg(file) {
    return new Promise(res => {
      const img = new Image();
      img.onload = () => {
        const c = document.createElement('canvas');
        const maxW = 794, maxH = 1123;
        let w = img.width, h = img.height;
        if (w > maxW || h > maxH) { const r = Math.min(maxW/w, maxH/h); w = Math.round(w*r); h = Math.round(h*r); }
        c.width = w; c.height = h;
        c.getContext('2d').drawImage(img, 0, 0, w, h);
        c.toBlob(b => {
          const fr = new FileReader();
          fr.onload = ev => res(ev.target.result.split(',')[1]);
          fr.readAsDataURL(b);
        }, 'image/jpeg', 0.88);
      };
      img.src = URL.createObjectURL(file);
    });
  }

  let _pdfDoc = null, _pdfPage = 1;

  async function renderPage(n) {
    const canvas = document.getElementById('pdf-cover-canvas');
    if (!_pdfDoc || !canvas) return;
    const page = await _pdfDoc.getPage(n);
    const vp0  = page.getViewport({ scale: 1 });
    const scale = 600 / vp0.width; // 300px × retina 2×
    const vp   = page.getViewport({ scale });
    canvas.width  = vp.width;
    canvas.height = vp.height;
    // reset animation so each page turn replays
    canvas.style.animation = 'none';
    canvas.offsetHeight; // reflow
    canvas.style.animation = '';
    await page.render({ canvasContext: canvas.getContext('2d'), viewport: vp }).promise;
    canvas.style.display = 'block';
    const info = document.getElementById('pdf-page-info');
    if (info) info.textContent = n + ' / ' + _pdfDoc.numPages;
    const prev = document.getElementById('pdf-prev');
    const next = document.getElementById('pdf-next');
    if (prev) prev.disabled = n <= 1;
    if (next) next.disabled = n >= _pdfDoc.numPages;
  }

  function changePage(delta) {
    if (!_pdfDoc) return;
    const next = _pdfPage + delta;
    if (next < 1 || next > _pdfDoc.numPages) return;
    _pdfPage = next;
    renderPage(_pdfPage);
  }

  async function renderPDFPreview(base64pdf) {
    const wrap = document.getElementById('pdf-preview-wrap');
    if (!window.pdfjsLib) return;
    try {
      const binaryStr = atob(base64pdf);
      const bytes = new Uint8Array(binaryStr.length);
      for (let i = 0; i < binaryStr.length; i++) bytes[i] = binaryStr.charCodeAt(i);
      _pdfDoc  = await pdfjsLib.getDocument({ data: bytes }).promise;
      _pdfPage = 1;
      await renderPage(1);
      if (wrap) wrap.style.display = 'block';
    } catch(e) { console.warn('[PDF preview]', e.message); }
  }

      if (!dragging) return;

  async function extractPDFText(file) {
    const arrayBuffer = await file.arrayBuffer();
    const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;
    let text = '';
    const maxPages = Math.min(pdf.numPages, 40);
    for (let i = 1; i <= maxPages; i++) {
      const page = await pdf.getPage(i);
      const content = await page.getTextContent();
      text += content.items.map(item => item.str).join(' ') + '\n';
    }
    return text.trim();
  }

  dnaFileInput.addEventListener('change', async e => {
    const file = e.target.files[0];
    if (!file) return;
    dnaChars.textContent = 'lendo...';
    try {
      let txt;
      if (file.name.toLowerCase().endsWith('.pdf')) {
        txt = await extractPDFText(file);
      } else {
        txt = await new Promise((res, rej) => {
          const reader = new FileReader();
          reader.onload = ev => res(ev.target.result);
          reader.onerror = rej;
          reader.readAsText(file, 'UTF-8');
        });
      }
      document.getElementById('dna-relatorio').value = txt;
      dnaChars.textContent = txt.length.toLocaleString('pt-BR') + ' chars carregados';
    } catch (err) {
      dnaChars.textContent = '❌ Erro ao ler arquivo';
    }
    e.target.value = '';
  });

      const data = JSON.parse(raw);
      AUTOSAVE_FIELDS.forEach(id => {
        const el = document.getElementById(id);
        if (el && data[id]) { el.value = data[id]; el.dispatchEvent(new Event('input')); }

        if (tc) tc.click();

      if (data._fmt) {

        document.getElementById('pv-sub').textContent = data['inp-subtitulo'];

  async function gerarCarrossel() {
    const assunto = document.getElementById('inp-assunto').value.trim();
    if (!assunto) { showErr('Informe o assunto do carrossel antes de gerar.'); return; }
    const payload = {
      tipo:             'carrossel',
      titulo:           assunto,
      subtitulo:        document.getElementById('inp-gancho').value.trim(),
      descricao:        document.getElementById('dna-nicho').value.trim() || assunto,
      autor:            document.getElementById('dna-nome').value.trim() || 'Nexus Digital',
      avatar:           document.getElementById('dna-publico').value.trim(),
      temaKey:          selectedTheme,
      formato:          carouselFmt,
      maxSlides:        parseInt(maxSliderEl?.value) || 6,
      formato_conteudo: selectedCtype,
      gancho:           document.getElementById('inp-gancho').value.trim(),
      ...(selectedFontTitle    && { fonteTitulo: selectedFontTitle }),
      ...(selectedFontBody     && { fonteCorpo:  selectedFontBody }),
    };
    const btnGerar = document.getElementById('btn-gerar');
    btnGerar.disabled = true;
    setCarouselProgress(0, 'Conectando ao servidor…');
    document.getElementById('overlay-carousel-prog').classList.add('show');
    hideErr();
    try {
      const initRes = await fetch('/api/carousel', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      });
      if (!initRes.ok) throw new Error('Falha ao iniciar (HTTP ' + initRes.status + ')');
      const { jobId } = await initRes.json();
      await new Promise((resolve, reject) => {
        const es = new EventSource('/api/carousel/progress/' + jobId);
        es.onmessage = e => {
          const d = JSON.parse(e.data);
          if (d.error) { es.close(); reject(new Error(d.error)); return; }
          if (d.progress !== undefined) setCarouselProgress(d.progress, d.message);
          if (d.done) { es.close(); showCarouselSuccess(d); resolve(d); }
        };
        es.onerror = () => { es.close(); reject(new Error('Conexão perdida — tente novamente.')); };
      });
    } catch (err) {
      document.getElementById('overlay-carousel-prog').classList.remove('show');
      showErr('⚠ ' + err.message);
    } finally {
      btnGerar.disabled = false;
    }
  }

  // ---- matrix rain ----
  (function(){
    const canvas = document.getElementById('matrix-canvas');
    const ctx    = canvas.getContext('2d');

    // Katakana + latin + numbers — classic matrix mix
    const CHARS = 'アイウエオカキクケコサシスセソタチツテトナニヌネノハヒフヘホマミムメモヤユヨラリルレロワヲンABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789@#$%&';

    const FONT_SIZE   = 14;
    const COLOR_HEAD  = '#FCD34D';    // bright amber — leading character (head of each stream)
    const COLOR_TRAIL = '#D97706';    // mid amber — trail body
    const COLOR_FADE  = '#92400E';    // dark amber — fading tail
    const SPEED_MIN   = 0.3;
    const SPEED_MAX   = 1.1;

    let cols, drops, speeds, chars, ages, maxAge;

    function resize() {
      canvas.width  = window.innerWidth;
      canvas.height = window.innerHeight;
      cols   = Math.floor(canvas.width / FONT_SIZE);
      maxAge = Math.floor(canvas.height / FONT_SIZE) + 10;
      drops  = Array.from({length: cols}, () => -Math.floor(Math.random() * maxAge));
      speeds = Array.from({length: cols}, () => SPEED_MIN + Math.random() * (SPEED_MAX - SPEED_MIN));
      chars  = Array.from({length: cols}, () =>
        Array.from({length: maxAge + 10}, () => CHARS[Math.floor(Math.random() * CHARS.length)])
      );
      ages   = Array.from({length: cols}, () => 0);
    }

    function draw() {
      // Fade background — low alpha so trails linger
      ctx.fillStyle = 'rgba(11,9,6,.18)';
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      ctx.font = `${FONT_SIZE}px "JetBrains Mono",monospace`;

      for (let i = 0; i < cols; i++) {
        const y = Math.floor(drops[i]) * FONT_SIZE;
        if (y < 0) { drops[i] += speeds[i]; continue; }

        // Occasionally mutate a character in the stream
        if (Math.random() < 0.04) {
          chars[i][Math.floor(Math.random() * chars[i].length)] =
            CHARS[Math.floor(Math.random() * CHARS.length)];
        }

        // Draw visible rows of each column stream
        const visibleRows = Math.min(Math.floor(drops[i]), maxAge);
        for (let r = 0; r < visibleRows; r++) {
          const rowY = (Math.floor(drops[i]) - r) * FONT_SIZE;
          if (rowY < 0 || rowY > canvas.height) continue;
          const ratio = r / Math.max(visibleRows, 1);
          if (r === 0) {
            ctx.fillStyle = COLOR_HEAD;
            ctx.shadowColor = COLOR_HEAD;
            ctx.shadowBlur  = 8;
          } else if (ratio < 0.25) {
            ctx.fillStyle = COLOR_TRAIL;
            ctx.shadowBlur = 3;
          } else {
            ctx.globalAlpha = Math.max(0, 1 - ratio * 1.6);
            ctx.fillStyle = COLOR_FADE;
            ctx.shadowBlur = 0;
          }
          ctx.fillText(chars[i][r % chars[i].length], i * FONT_SIZE, rowY);
          ctx.globalAlpha = 1;
          ctx.shadowBlur  = 0;
        }

        // Advance drop; reset when it falls off screen
        drops[i] += speeds[i];
        if (drops[i] * FONT_SIZE > canvas.height + FONT_SIZE * 20) {
          drops[i] = -Math.floor(Math.random() * maxAge * 0.6);
          speeds[i] = SPEED_MIN + Math.random() * (SPEED_MAX - SPEED_MIN);
        }
      }
    }

    resize();
    window.addEventListener('resize', resize);

    let last = 0;
    const FPS = 22;
    function loop(ts) {
      if (ts - last > 1000 / FPS) { draw(); last = ts; }
      requestAnimationFrame(loop);
    }
    requestAnimationFrame(loop);
  })();

  // ── CONVERSOR PDF ↔ DOCX ──────────────────────────────────────────────────
  (function() {
    const drop      = document.getElementById('conv-drop');
    const fileInput = document.getElementById('conv-file-input');
    const pickBtn   = document.getElementById('conv-pick-btn');
    const options   = document.getElementById('conv-options');
    const fileName  = document.getElementById('conv-file-name');
    const fileSize  = document.getElementById('conv-file-size');
    const removeBtn = document.getElementById('conv-remove-btn');
    const dirCards  = document.querySelectorAll('.conv-dir-card');
    const convBtn   = document.getElementById('conv-btn');
    const progressW = document.getElementById('conv-progress');
    const progressF = document.getElementById('conv-progress-fill');
    const statusMsg = document.getElementById('conv-status-msg');
    const resultDiv = document.getElementById('conv-result');
    const dlBtn     = document.getElementById('conv-download-btn');
    const resetBtn  = document.getElementById('conv-reset-btn');

    let convFile = null;
    let convDirection = 'docx2pdf'; // default
    let convResultData = null;

    function fmt(bytes) {
      if (bytes < 1024) return bytes + ' B';
      if (bytes < 1024*1024) return (bytes/1024).toFixed(1) + ' KB';
      return (bytes/(1024*1024)).toFixed(1) + ' MB';
    }

    const dropInner = document.getElementById('conv-drop-inner');

    function setFile(file) {
      if (!file) return;
      if (file.size > 10 * 1024 * 1024) { showErr('Arquivo muito grande. Máximo 10MB.'); return; }
      const ext = file.name.split('.').pop().toLowerCase();
      if (!['pdf','docx'].includes(ext)) { showErr('Formato inválido. Use .pdf ou .docx'); return; }
      convFile = file;
      fileName.textContent = file.name;
      fileSize.textContent = fmt(file.size);

      // Atualiza drop zone para mostrar arquivo selecionado
      const icon = ext === 'pdf' ? '📕' : '📝';
      dropInner.innerHTML = `
        <span class="conv-big-icon">${icon}</span>
        <p class="conv-drop-title" style="font-size:15px">${file.name}</p>
        <p class="conv-drop-sub">${fmt(file.size)} · clique para trocar</p>
      `;
      drop.style.minHeight = '120px';
      drop.classList.add('has-file');

      // Auto-detectar direção pelo tipo do arquivo
      if (ext === 'docx') {
        convDirection = 'docx2pdf';
        document.getElementById('cdir-pdf').classList.add('active');
        document.getElementById('cdir-docx').classList.remove('active');
        document.querySelector('#cdir-pdf b').textContent = '→ PDF';
        document.querySelector('#cdir-pdf small').textContent = 'Word → PDF (atual)';
        document.querySelector('#cdir-docx small').textContent = 'PDF → Word (invertido)';
      } else {
        convDirection = 'pdf2docx';
        document.getElementById('cdir-docx').classList.add('active');
        document.getElementById('cdir-pdf').classList.remove('active');
        document.querySelector('#cdir-docx b').textContent = '→ Word';
        document.querySelector('#cdir-docx small').textContent = 'PDF → Word (atual)';
        document.querySelector('#cdir-pdf small').textContent = 'Word → PDF (invertido)';
      }

      options.style.display = 'block';
      resultDiv.style.display = 'none';
      progressW.style.display = 'none';
      statusMsg.textContent = '';
    }

    function resetConv() {
      convFile = null;
      convResultData = null;
      drop.classList.remove('has-file');
      drop.style.minHeight = '';
      drop.style.display = '';
      options.style.display = 'none';
      resultDiv.style.display = 'none';
      progressW.style.display = 'none';
      statusMsg.textContent = '';
      progressF.style.width = '0%';
      fileInput.value = '';
      dropInner.innerHTML = `
        <span class="conv-big-icon">📄</span>
        <p class="conv-drop-title">Arraste seu arquivo aqui</p>
        <p class="conv-drop-sub">PDF ou Word (.docx) · máx. 10MB</p>
        <button class="btn" style="pointer-events:all">Escolher arquivo</button>
      `;
      // Reset direction cards
      document.getElementById('cdir-pdf').classList.add('active');
      document.getElementById('cdir-docx').classList.remove('active');
      document.querySelector('#cdir-pdf b').textContent = '→ PDF';
      document.querySelector('#cdir-pdf small').textContent = 'Word para PDF';
      document.querySelector('#cdir-docx b').textContent = '→ Word';
      document.querySelector('#cdir-docx small').textContent = 'PDF para Word';
      convDirection = 'docx2pdf';
    }

    // Drag & drop
    drop.addEventListener('dragover', e => { e.preventDefault(); drop.classList.add('over'); });
    drop.addEventListener('dragleave', () => drop.classList.remove('over'));
    drop.addEventListener('drop', e => {
      e.preventDefault();
      drop.classList.remove('over');
      const f = e.dataTransfer?.files?.[0];
      if (f) setFile(f);
    });
    // Clicar na drop zone abre o seletor sempre (inclusive para trocar arquivo)
    drop.addEventListener('click', e => {
      if (e.target === removeBtn) return;
      fileInput.click();
    });
    pickBtn.addEventListener('click', e => { e.stopPropagation(); fileInput.click(); });
    fileInput.addEventListener('change', () => { if (fileInput.files[0]) setFile(fileInput.files[0]); });
    removeBtn.addEventListener('click', e => { e.stopPropagation(); resetConv(); });

    // Direction selector
    dirCards.forEach(card => {
      card.addEventListener('click', () => {
        dirCards.forEach(c => c.classList.remove('active'));
        card.classList.add('active');
        convDirection = card.dataset.dir === 'auto-pdf' ? 'docx2pdf' : 'pdf2docx';
      });
    });

    // Convert button
    convBtn.addEventListener('click', async () => {
      if (!convFile) return;
      convBtn.disabled = true;
      progressW.style.display = 'block';
      statusMsg.textContent = 'Lendo arquivo...';
      progressF.style.width = '20%';

      try {
        const base64 = await new Promise((res, rej) => {
          const reader = new FileReader();
          reader.onload = () => res(reader.result.split(',')[1]);
          reader.onerror = rej;
          reader.readAsDataURL(convFile);
        });

        statusMsg.textContent = 'Convertendo...';
        progressF.style.width = '55%';

        const resp = await fetch('/api/convert', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ arquivo: base64, nome: convFile.name, direcao: convDirection }),
        });

        progressF.style.width = '85%';
        if (!resp.ok) {
          const err = await resp.json();
          throw new Error(err.error || 'Erro na conversão');
        }

        convResultData = await resp.json();
        progressF.style.width = '100%';
        statusMsg.textContent = '';

        setTimeout(() => {
          progressW.style.display = 'none';
          drop.style.display = 'none';
          options.style.display = 'none';
          resultDiv.style.display = 'block';
        }, 400);

      } catch (e) {
        progressW.style.display = 'none';
        statusMsg.textContent = '';
        showErr('Falha na conversão: ' + e.message);
        convBtn.disabled = false;
      }
    });

    // Download
    dlBtn.addEventListener('click', () => {
      if (!convResultData) return;
      const a = document.createElement('a');
      a.href = `data:${convResultData.tipo};base64,${convResultData.arquivo}`;
      a.download = convResultData.nome;
      a.click();
    });

    resetBtn.addEventListener('click', resetConv);
  })();
</script>
